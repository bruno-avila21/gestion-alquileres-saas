<objective>
Build the Infrastructure-layer HTTP clients that talk to BCRA and INDEC, with resilience
handled by `Microsoft.Extensions.Http.Resilience 8.10.0` (NOT the deprecated Polly package).

the SyncIndexCommandHandler fallback path (IDX-04).

mock `HttpMessageHandler` (no real network calls in CI) to verify date parsing, endpoint paths,
and error propagation.

`DependencyInjection.cs`, which this plan touches to register HttpClients while Plan 01
registers the repository. Both plans must be merged before Plan 03 runs; conflict resolution
on DependencyInjection.cs is trivial (both add independent registration lines).

    ```csharp
    using System.Globalization;
    using System.Net.Http.Json;
    using Microsoft.Extensions.Logging;

    namespace GestionAlquileres.Infrastructure.ExternalServices;

    public class BcraApiClient
    {
        private readonly HttpClient _httpClient;
        private readonly ILogger<BcraApiClient> _logger;

        // BCRA variable ID for ICL (Índice para Contratos de Locación).
        // Verified via community code — re-confirm via /estadisticas/v1/principalesvariables if API changes.
        public const int IclVariableId = 7988;

        public BcraApiClient(HttpClient httpClient, ILogger<BcraApiClient> logger)
        {
            _httpClient = httpClient;
            _logger = logger;
        }

        public async Task<IReadOnlyList<BcraDataPoint>> GetIclAsync(
            DateOnly desde, DateOnly hasta, CancellationToken ct = default)
        {
            // v1 path — confirmed by multiple community implementations.
            var url = $"/estadisticas/v1/datosvariable/{IclVariableId}" +
                      $"/{desde:yyyy-MM-dd}/{hasta:yyyy-MM-dd}";

            _logger.LogInformation("Fetching BCRA ICL from {Url}", url);

            var raw = await _httpClient.GetFromJsonAsync<BcraRawResponse>(url, ct);
            if (raw?.Results is null || raw.Results.Count == 0)
                return Array.Empty<BcraDataPoint>();

            var result = new List<BcraDataPoint>(raw.Results.Count);
            foreach (var point in raw.Results)
            {
                // RESEARCH Pitfall 1: dd/MM/yyyy, not MM/dd/yyyy.
                var fecha = DateOnly.ParseExact(point.Fecha, "dd/MM/yyyy", CultureInfo.InvariantCulture);
                result.Add(new BcraDataPoint(fecha, point.Valor));
            }
            return result;
        }
    }
    ```

    ```csharp
    using System.Globalization;
    using System.Net.Http.Json;
    using Microsoft.Extensions.Logging;

    namespace GestionAlquileres.Infrastructure.ExternalServices;

    public class IndecApiClient
    {
        private readonly HttpClient _httpClient;
        private readonly ILogger<IndecApiClient> _logger;

        // Series ID for IPC Nivel General (base diciembre 2016 = 100).
        public const string IpcSeriesId = "148.3_INIVELNAL_DICI_M_26";

        public IndecApiClient(HttpClient httpClient, ILogger<IndecApiClient> logger)
        {
            _httpClient = httpClient;
            _logger = logger;
        }

        public async Task<IReadOnlyList<IndecDataPoint>> GetIpcAsync(
            DateOnly desde, DateOnly hasta, CancellationToken ct = default)
        {
            var url = $"/series/api/series/?ids={IpcSeriesId}" +
                      $"&start_date={desde:yyyy-MM-dd}&end_date={hasta:yyyy-MM-dd}&limit=1000";

            _logger.LogInformation("Fetching INDEC IPC from {Url}", url);

            var raw = await _httpClient.GetFromJsonAsync<IndecRawResponse>(url, ct);
            if (raw?.Data is null || raw.Data.Count == 0)
                return Array.Empty<IndecDataPoint>();

            var result = new List<IndecDataPoint>(raw.Data.Count);
            foreach (var row in raw.Data)
            {
                if (row.Length < 2) continue;
                var dateStr = row[0].GetString();
                if (string.IsNullOrWhiteSpace(dateStr)) continue;
                var fecha = DateOnly.ParseExact(dateStr, "yyyy-MM-dd", CultureInfo.InvariantCulture);
                var valor = row[1].GetDecimal();
                result.Add(new IndecDataPoint(fecha, valor));
            }
            return result;
        }
    }
    ```

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Application → Infrastructure HTTP client | Application trusts client to validate response schema |
| Infrastructure HTTP client → BCRA/INDEC public APIs | Untrusted remote — outage, malformed data, slow responses all plausible |
| DI configuration → runtime | Base URLs are hard-coded in DI (not from config), preventing SSRF |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-02-06 | Tampering (SSRF) | BcraApiClient base URL | mitigate | Base URL hardcoded in DependencyInjection.cs — NEVER taken from user input. No configuration binding of URL. |
| T-02-07 | Denial of Service | External API outage cascading to our thread pool | mitigate | `AddStandardResilienceHandler` applies circuit breaker (opens on >10% failure rate) and total request timeout (60s) — prevents thread exhaustion. |
| T-02-08 | Tampering | Malformed response (non-numeric valor, wrong date format) | mitigate | DateOnly.ParseExact with invariant culture throws on bad format — surfaces as FormatException to handler, triggering fallback path (IDX-04). |
| T-02-09 | Information Disclosure | Request logs may contain full URL with ID | accept | URL contains only public identifiers (7988, series ID) — no secrets. Still log at Information level, not Warning. |
| T-02-10 | Denial of Service | Retry storm against BCRA during their outage | mitigate | AddStandardResilienceHandler default retry uses exponential backoff with jitter — avoids synchronized retries across instances. Max 3 attempts cap total load. |
| T-02-11 | Spoofing | TLS downgrade / rogue endpoint | mitigate | HTTPS enforced via `Uri("https://...")`. .NET 8 default cert validation applies — do NOT disable. |
</threat_model>
